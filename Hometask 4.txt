import Foundation

var optionalVariable: Int? = 10

print("Переменная = \(optionalVariable!), но если nil- был бы краш")        // (1)

if optionalVariable != nil {                                                  // (2)
  print("Variable = \(optionalVariable!)")
} else {
  print("It's empty")
}

if let tempVariable = optionalVariable {                                      // (3)
  print("Переменная = \(tempVariable)")
} else {
  print("Переменная не введена")
}

let unionVariable = optionalVariable ?? 0                                     // (4)

print("3 способ - \(unionVariable)")
